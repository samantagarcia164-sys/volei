    script
        let jogadores = JSON.parse(localStorage.getItem('jogadoresVolei'))  [];

        function renderizar() {
            const lista = document.getElementById('listaJogadores');
            lista.innerHTML = '';

            jogadores.forEach((j, index) = {
                const item = document.createElement('li');
                item.className = `flex justify-between items-center p-3 rounded-lg border ${j.pago  'bg-green-50 border-green-200'  'bg-red-50 border-red-200'}`;
                
                item.innerHTML = `
                    span class=font-medium ${j.pago  'text-green-800'  'text-red-800'}${j.nome}span
                    div class=flex gap-2
                        button onclick=togglePix(${index}) 
                            class=text-xs px-3 py-1 rounded-full font-bold uppercase ${j.pago  'bg-green-500 text-white'  'bg-red-400 text-white'}
                            ${j.pago  'PIX OK'  'Pendente'}
                        button
                        button onclick=removerJogador(${index}) class=text-gray-400 hovertext-red-600✕button
                    div
                `;
                lista.appendChild(item);
            });
            
            localStorage.setItem('jogadoresVolei', JSON.stringify(jogadores));
        }

        function adicionarJogador() {
            const input = document.getElementById('nomeJogador');
            if (input.value.trim() !== '') {
                jogadores.push({ nome input.value, pago false });
                input.value = '';
                renderizar();
            }
        }

        function togglePix(index) {
            jogadores[index].pago = !jogadores[index].pago;
            renderizar();
        }

        function removerJogador(index) {
            jogadores.splice(index, 1);
            renderizar();
        }

        renderizar();
    script
